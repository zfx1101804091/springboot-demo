#development
#log4j.rootLogger=ERROR,file
#log4j.logger.com.bluecardsoft.park=DEBUG,stdout,dfile
log4j.rootLogger=DEBUG
log4j.logger.com.example.springbootshiro=DEBUG,stdout
#log4j.logger.com.blue=DEBUG,stdout,dfile

log4j.appender.file=com.example.springbootshiro.utils.MyDailyRollingFileAppender
#Windows:
#log4j.appender.file.File=${catalina.home}/webapps/main/logs/error/error
#Linux:
log4j.appender.file.File=/zh/logs/error/error
log4j.appender.file.DatePattern='.'yyyy-MM-dd'.zip'
log4j.appender.file.Append = true 
log4j.appender.file.Encoding = UTF-8
log4j.appender.file.layout=org.apache.log4j.PatternLayout
log4j.appender.file.layout.ConversionPattern=%d{yyyy-MM-dd HH:mm:ss.SSS}: [%p] [%t] [%c:%L] %x %m%n
        

### direct log messages to stdout ###
log4j.appender.stdout=org.apache.log4j.ConsoleAppender
log4j.appender.stdout.Target=System.out
log4j.appender.stdout.layout=org.apache.log4j.PatternLayout
log4j.appender.stdout.layout.ConversionPattern=%d{ABSOLUTE} %5p [%t] [%c]->[%F]:%L - %m%n


### controller###
log4j.logger.com.example.springbootshiro.controller= DEBUG, controller
log4j.appender.controller=com.example.springbootshiro.utils.MyDailyRollingFileAppender
log4j.appender.controller.File=/zh/logs/controller/controller
log4j.appender.controller.DatePattern='.'yyyyMMdd_HH'.zip'
log4j.appender.controller.Append = true
log4j.appender.ticket.Encoding = UTF-8 
log4j.appender.controller.layout=org.apache.log4j.PatternLayout
log4j.appender.controller.layout.ConversionPattern=%d{HH\:mm\:ss.SSS} %t [%c]- %m%n
